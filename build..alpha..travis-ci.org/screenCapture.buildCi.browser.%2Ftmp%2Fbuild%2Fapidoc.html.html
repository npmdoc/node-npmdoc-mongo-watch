    function MongoWatch(options) {
  this.options = applyDefaults(options);
  this.channel = new EventEmitter;
  this.channel.on('error', this.options.onError);
  this.channel.on('connected', (function(_this) {
    return function() {
      return _this.status = 'connected';
    };
  })(this));
  this.debug = this.options.onDebug;
  getOplogStream(this.options, (function(_this) {
    return function(err, stream, oplogClient) {
      _this.stream = stream;
      _this.oplogClient = oplogClient;
      if (err) {
        _this.channel.emit('error', err);
      }
      _this.debug("Emiting 'connected'. Stream exists:", _this.stream != null);
      return _this.channel.emit('connected');
    };
  })(this));
}
